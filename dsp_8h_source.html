66{

101{
102 float frac;
103 int exp;
104 frac = frexpf(fabsf(f), &exp);
105 f = 1.23149591368684f;
106 f *= frac;
107 f += -4.11852516267426f;
108 f *= frac;
109 f += 6.02197014179219f;

160inline float
161fmap(float in, float min, float max, Mapping curve = Mapping::LINEAR)
162{
163 switch(curve)
164 {
165 case Mapping::EXP:
166 return fclamp(min + (in * in) * (max - min), min, max);
167 case Mapping::LOG:
168 {
169 const float a = 1.f / log10f(max / min);
170 return fclamp(min * powf(10, in / a), min, max);
171 }
172 case Mapping::LINEAR:
173 default: return fclamp(min + in * (max - min), min, max);
174 }
175}
